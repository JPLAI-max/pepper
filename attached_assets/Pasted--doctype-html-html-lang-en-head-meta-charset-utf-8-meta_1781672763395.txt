<script>
  const DISCLAIM = "Simulation only. All loans, capital, rates, prices, balances, tokens, and digital assets shown are illustrative and do not represent real instruments, offerings, funding, trades, or settlement. No money or value moves. Built as a prototype to illustrate product scope and surface regulatory questions for legal review.";
  ['disc1','disc2','disc3'].forEach(id=>document.getElementById(id).textContent=DISCLAIM);
  const usd = n => "$"+Math.round(n).toLocaleString();
  let LOANS = [
    { id:"L1", name:"Bridge · Maple Court Triplex", loc:"Austin, TX", amt:420000, yield:8.2, ltv:68, term:"12 mo", funded:0 },
    { id:"L2", name:"DSCR · Riverside 18-unit", loc:"Phoenix, AZ", amt:1250000, yield:7.4, ltv:72, term:"30 yr", funded:0 },
    { id:"L3", name:"Fix & Flip · Oakwood Duplex", loc:"Nashville, TN", amt:310000, yield:9.1, ltv:70, term:"9 mo", funded:0 },
    { id:"L4", name:"Bridge · Harbor View Lofts", loc:"Tampa, FL", amt:880000, yield:7.0, ltv:64, term:"18 mo", funded:0 },
    { id:"L5", name:"DSCR · Sunset Bungalows", loc:"San Antonio, TX", amt:540000, yield:8.6, ltv:74, term:"30 yr", funded:0 },
    { id:"L6", name:"Fix & Flip · Cedar Park Homes", loc:"Raleigh, NC", amt:260000, yield:9.4, ltv:78, term:"9 mo", funded:0 },
  ];
  let funded = [];
  function renderLoans(){
    document.getElementById('loanList').innerHTML = LOANS.map(l=>{
      const pct = Math.round(l.funded/l.amt*100);
      const done = l.funded>=l.amt;
      return `<div class="loan" data-id="${l.id}">
        <div><div class="name">${l.name}</div><div class="sub">${l.loc} · ${l.term}</div></div>
        <div class="col"><div class="k">Loan amount</div><div class="v">${usd(l.amt)}</div></div>
        <div class="col"><div class="k">Yield</div><div class="v" style="color:var(--gold)">${l.yield}%</div></div>
        <div class="col"><div class="k">LTV</div><div class="v">${l.ltv}%</div></div>
        <div class="col"><div class="k">Funded</div><div class="v">${pct}%</div></div>
        ${done?'<span class="funded-tag">✓ Funded</span>':`<button class="fundbtn" data-id="${l.id}">Table-fund</button>`}
        <div class="progress"><span style="width:${pct}%"></span></div>
      </div>`;
    }).join('');
    document.querySelectorAll('.fundbtn').forEach(b=>b.onclick=()=>fundLoan(b.dataset.id));
  }
  function fundLoan(id){ const l = LOANS.find(x=>x.id===id); if(!l) return; l.funded = l.amt; moveToSecondary(l); renderLoans(); }
  function moveToSecondary(l){ if(funded.find(f=>f.id===l.id)) return; funded.push({ id:l.id, name:l.name, loc:l.loc, yield:l.yield, price:100.0, chg:(Math.random()*2-0.6) }); }
  document.getElementById('runBtn').onclick=()=>{
    let cap = +document.getElementById('cap').value||0;
    const minY = +document.getElementById('miny').value||0;
    const maxL = +document.getElementById('maxltv').value||100;
    const maxPer = +document.getElementById('maxper').value||cap;
    let deployed=0, count=0, yields=[];
    LOANS.filter(l=>l.funded<l.amt && l.yield>=minY && l.ltv<=maxL)
         .sort((a,b)=>b.yield-a.yield)
         .forEach(l=>{
            if(cap<=0) return;
            const need = l.amt - l.funded;
            const alloc = Math.min(need, maxPer, cap);
            if(alloc<=0) return;
            l.funded += alloc; cap-=alloc; deployed+=alloc; count++; yields.push(l.yield);
            if(l.funded>=l.amt) moveToSecondary(l);
         });
    const avg = yields.length? (yields.reduce((a,b)=>a+b,0)/yields.length).toFixed(1) : 0;
    document.getElementById('engineResult').textContent = count
      ? `Auto-funded ${usd(deployed)} across ${count} loan${count>1?'s':''} at ${avg}% avg yield. ${usd(cap)} capital remaining. (Simulated)`
      : `No loans matched your criteria. (Simulated)`;
    renderLoans();
  };
  function renderSecondary(){
    const el = document.getElementById('smList');
    el.innerHTML = funded.length ? funded.map(f=>`
      <div class="sm">
        <div><div class="name">${f.name}</div><div class="sub">${f.loc} · funded loan position</div></div>
        <div class="col"><div class="k">Price / $100</div><div class="v">$${f.price.toFixed(2)}</div></div>
        <div class="col"><div class="k">Yield</div><div class="v" style="color:var(--gold)">${f.yield}%</div></div>
        <div class="col ${f.chg>=0?'up':'down'}"><div class="k">Today</div><div class="v">${f.chg>=0?'▲':'▼'} ${Math.abs(f.chg).toFixed(2)}%</div></div>
        <button class="trade">Trade</button>
      </div>`).join('')
      : `<div class="subtitle">No funded positions yet — fund a loan on the Funding Desk and it appears here as a tradeable position.</div>`;
  }
  const TOKENS = [
    { sym:"pUSD", name:"Pepper USD (stablecoin)", sub:"Settlement rail · concept", bal:"12,500.00" },
    { sym:"INC", name:"Tokenized Income Unit", sub:"Maple Court Triplex · concept", bal:"120 units" },
    { sym:"LN", name:"Tokenized Loan Position", sub:"Oakwood Duplex · concept", bal:"$3,100" },
  ];
  function renderTokens(){
    document.getElementById('tokenList').innerHTML = TOKENS.map(t=>`
      <div class="token"><div class="tok-ic">${t.sym}</div>
      <div><div class="name">${t.name}</div><div class="sub">${t.sub}</div></div>
      <div class="bal">${t.bal}</div></div>`).join('');
  }
  function show(tab){
    document.querySelectorAll('.view').forEach(v=>v.classList.remove('on'));
    document.getElementById('v-'+tab).classList.add('on');
    document.querySelectorAll('.tabs button').forEach(b=>b.classList.toggle('on', b.dataset.tab===tab));
    if(tab==='secondary') renderSecondary();
    if(tab==='crypto') renderTokens();
    window.scrollTo(0,0);
  }
  document.getElementById('t-funding').onclick=()=>show('funding');
  document.getElementById('t-secondary').onclick=()=>show('secondary');
  document.getElementById('t-crypto').onclick=()=>show('crypto');
  renderLoans();
</script>
