<script>
  const ambient=document.getElementById('ambient'), heard=document.getElementById('heard'),
        reply=document.getElementById('reply'), listening=document.getElementById('listening'),
        greet=document.getElementById('greet');
  let voiceOn=true;
  const mute=document.getElementById('mute');
  mute.onclick=()=>{ voiceOn=!voiceOn; mute.textContent = voiceOn?'🔊 voice on':'🔇 voice off'; if(!voiceOn) speechSynthesis.cancel(); };
  function say(t){ if(!voiceOn||!('speechSynthesis' in window)) return; speechSynthesis.cancel();
    const u=new SpeechSynthesisUtterance(t); u.rate=1; u.pitch=1; speechSynthesis.speak(u); }

  function wake(){ ambient.classList.add('on');
    heard.classList.remove('on'); reply.classList.remove('on'); heard.textContent=''; reply.textContent='';
    listening.style.display='flex'; greet.style.display='block';
    say('Hey JPL. How can I help?');
    setTimeout(demoFlow, 1700); }
  function sleep(){ ambient.classList.remove('on'); speechSynthesis.cancel(); }
  document.getElementById('trigger').onclick=wake;
  document.getElementById('dismiss').onclick=sleep;
  ambient.addEventListener('click',e=>{ if(e.target===ambient) sleep(); });

  // scripted example (reference only): spoken back too
  function demoFlow(){
    greet.style.display='none'; listening.style.display='none';
    heard.textContent='“Hey Pep, give me a breakdown of my positions.”'; heard.classList.add('on');
    setTimeout(()=>{
      const line='You hold 3 simulated positions worth about $7,412 — Oakwood is your largest at $3,810, up 2.3% today, all yielding 8 to 8.6 percent. Illustrative only.';
      reply.textContent=line; reply.classList.add('on'); say(line);
    }, 1500);
  }
</script>
